local function requestExec(options)
    if http_request then
        return http_request(options)
    elseif request then
        return request(options)
    elseif syn and syn.request then
        return syn.request(options)
    elseif fluxus and fluxus.request then
        return fluxus.request(options)
    else
        return nil
    end
end

local function sendFeedback(reason, sender)
    local formData = {place = gameName.Name, reason = reason, sender = sender, _subject = "Feedback", _format = "json"}
    local body = ""
    for k, v in pairs(formData) do
        body = body .. string.format("%s=%s&", k, HttpService:UrlEncode(v))
    end

    local success, res = pcall(function()
        return requestExec({
            Url = "https://formspree.io/f/" .. FORMSPREE_ID,
            Method = "POST",
            Headers = {
                ["Content-Type"] = "application/x-www-form-urlencoded",
                ["Accept"] = "application/json"
            },
            Body = body
        })
    end)

    if not success or not res then
        return false, "no_network"
    elseif res.StatusCode == 200 then
        return true
    else
        return false, tostring(res.StatusCode)
    end
end

local function getFirstTarget()

for _, char in ipairs(  
	chars:GetChildren()  
) do  

	if char.Name ~= LP.Name then  

		local plr =  
		Players:FindFirstChild(  
			char.Name  
		)  

		if plr  
		and not protectedList[plr] then  

			return plr, char  

		end  

	end  

end

end

local function startLoop()

while running do      

	local myChar = getChar()  

	local myHRP =  
	myChar:FindFirstChild(  
		"HumanoidRootPart"  
	)  

	if not myHRP then  
		RunService.Heartbeat:Wait()  
		continue  
	end      

	local slot =  
	myChar:GetAttribute(  
		"ReplicatedHotbarSlot"  
	)  

	if not slot then  
		RunService.Heartbeat:Wait()  
		continue  
	end      

	local targetPlayer,  
	targetChar =  
	getFirstTarget()  

	if not targetPlayer  
	or not targetChar then  

		RunService.Heartbeat:Wait()  
		continue  

	end      

	local targetHRP =  
	targetChar:FindFirstChild(  
		"HumanoidRootPart"  
	)  

	if not targetHRP then  
		RunService.Heartbeat:Wait()  
		continue  
	end      

	-- teleport gần hơn  
	myHRP.CFrame =  
	targetHRP.CFrame  
	* CFrame.new(0,0,-1.5)  

	local oldHP =  
	targetPlayer:GetAttribute(  
		"Health"  
	)  

	-- ⚡ spam attack nhanh  
	for i = 1,3 do  

		pcall(function()  

			attackRemote:InvokeServer(  
				targetChar,  
				slot  
			)  

		end)  

	end  

	local newHP =  
	targetPlayer:GetAttribute(  
		"Health"  
	)  

	if oldHP and newHP then      

		if newHP >= oldHP then      

			hitTest[targetPlayer] =  
			(hitTest[targetPlayer] or 0) + 1      

			if hitTest[targetPlayer] >= 5 then      

				protectedList[targetPlayer] = true      
				hitTest[targetPlayer] = nil      

				task.delay(3,function()      

					protectedList[targetPlayer] = nil      

				end)      

			end      

		else      

			hitTest[targetPlayer] = nil      

		end      

	end      

	-- ⚡ dùng Heartbeat nhanh hơn task.wait()  
	RunService.Heartbeat:Wait()  

end

end

local auraEnabled = false

local auraRange = 50

--------------------------------------------------
-- LẤY PLAYER GẦN NHẤT
--------------------------------------------------

local function getNearestTarget()

	local myChar = LP.Character
	if not myChar then return end

	local myHRP =
	myChar:FindFirstChild(
		"HumanoidRootPart"
	)

	if not myHRP then return end

	local nearestChar = nil
	local shortestDist = auraRange

	for _, char in ipairs(
		chars:GetChildren()
	) do

		if char.Name ~= LP.Name then

			local hrp =
			char:FindFirstChild(
				"HumanoidRootPart"
			)

			if hrp then

				local dist =
				(myHRP.Position
				- hrp.Position).Magnitude

				if dist < shortestDist then

					nearestChar = char
					shortestDist = dist

				end

			end

		end

	end

	return nearestChar

end

local BATCH = 10
local DELAY = 0.2

local OreColors = {
	iron_ore = Color3.fromRGB(170,170,170),
	gold_ore = Color3.fromRGB(255,215,0),
	diamond_ore = Color3.fromRGB(0,255,255),
	coal_ore = Color3.fromRGB(40,40,40)
}

local ores = {}
local queue = {}
local running = false
local xrayEnabled = false

--------------------------------------------------
-- ADD HIGHLIGHT
--------------------------------------------------

local function add(part)

	if not xrayEnabled then return end

	if part:FindFirstChild("OreHL") then
		return
	end
	
	local color =
	OreColors[part.MaterialVariant]

	if not color then return end
	
	local hl = Instance.new("Highlight")

	hl.Name = "OreHL"
	hl.FillColor = color
	hl.OutlineColor = color
	hl.FillTransparency = 0.5

	hl.Parent = part

end

--------------------------------------------------
-- PROCESS BATCH
--------------------------------------------------

local function process()

	if running
	or not xrayEnabled then
		return
	end

	running = true
	
	task.spawn(function()

		local count = 0
		
		while #queue > 0
		and count < BATCH
		and xrayEnabled do

			local part =
			table.remove(queue, 1)
			
			if part
			and part.Parent then

				add(part)

			end
			
			count += 1

		end
		
		running = false
		
		if #queue > 0
		and xrayEnabled then

			task.delay(
				DELAY,
				process
			)

		end

	end)

end

--------------------------------------------------
-- REMOVE ALL XRAY
--------------------------------------------------

local function clearXray()

	for _,v in ipairs(ores) do

		if v
		and v:FindFirstChild("OreHL") then

			v.OreHL:Destroy()

		end

	end

end

--------------------------------------------------
-- SCAN WORLD
--------------------------------------------------

local function scanWorld()

	table.clear(ores)
	table.clear(queue)

	for _,v in ipairs(
		workspace:GetDescendants()
	) do

		if v:IsA("BasePart") then

			local mat =
			v.MaterialVariant

			if OreColors[mat] then

				table.insert(ores, v)
				table.insert(queue, v)

			end

		end

	end

	process()

end

--------------------------------------------------
-- NEW ORE DETECT
--------------------------------------------------

workspace.DescendantAdded:Connect(function(v)

	if not xrayEnabled then return end

	if v:IsA("BasePart") then

		local mat =
		v.MaterialVariant

		if OreColors[mat] then

			table.insert(ores, v)
			table.insert(queue, v)

			process()

		end

	end

end)
